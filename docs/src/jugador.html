136        """
137        Funcion actualizar:

150                    continue

152    def jugar_cartas(self):
153        """
154        Funcion jugar cartas:
155            Comprueba si hay cartas seleccionadas y las juega o las descarta.
156        """
157        if (
158            len(self._cartas_seleccionadas) <= self.limite_seleccion
159            and self._cartas_seleccionadas
160        ):  # Comprueba que haya cartas seleccionadas
161            self.limite_jugar -= 1
162
163            evaluacion = self.evaluador.evaluar(self._cartas_seleccionadas)
164            self._cartas_jugadas = evaluacion["Cartas"]
165            self.puntos_cartas = evaluacion["Valor"]
166            self.multiplicador = evaluacion["Multiplicador"]
167
168            if self.comodines_mano:
169                for comodin in self.comodines_mano:
170                    self.puntos_cartas, self.multiplicador, self.dinero = (
171                        comodin.aplicar(
172                            self.mano,
173                            self.puntos_cartas,
174                            self.multiplicador,
175                            self.dinero,
176                            self._cartas_jugadas,
177                            self.comodines_mano,
178                        )
179                    )
180
181                # print(f"Operacion despues: {self.puntos_cartas} * {self.multiplicador} = {self.puntos}")
182
183            self.puntos += self.puntos_cartas * self.multiplicador
184
185            for (
186                carta
187            ) in (
188                self._cartas_jugadas
189            ):  # Quita las cartas que se hayan jugado para que el resto se descarte correctamente
190                self.mano.remove(carta)
191                self._cartas_seleccionadas.remove(carta)
192            self.descartar_cartas(
193                "jugar_cartas"
194            )  # Las cartas que no se hayan jugado se descartan
195            self.animador_texto.iniciar(
196                f"{self.puntos_cartas * self.multiplicador}", 1100, 150, y_final=90
197            )
198
199            if self.puntos >= self.puntos_nivel:
200                self.siguiente_ronda()
201            elif self.limite_jugar <= 0:
202                self.game_over = True

204    def descartar_cartas(self, origen):
205        """
206        Funcion descartar cartas:
207            Descarta las cartas del mazo.
208        """
209        if origen == "boton" and self.limite_descartar > 0:
210            self.limite_descartar -= 1
211        elif not origen == "jugar_cartas":
212            return None
213
214        self._cartas_descartadas.clear()
215        if (
216            len(self._cartas_seleccionadas) <= self.limite_seleccion
217            and self._cartas_seleccionadas
218        ):
219            for carta in self._cartas_seleccionadas:
220                self._cartas_descartadas.append(carta)
221                self.mano.remove(carta)
222            self._cartas_seleccionadas.clear()
223        while len(self.mano) != 8:
224            self.mano.append(self.mazo.robar())

255        Funcion cargar partida:

266
267            self.limite_descartar = datos["limite_descartar"]
268            self.limite_jugar = datos["limite_jugar"]
269
270            self.comodines_mano = (
271                [Comodin(nombre) for nombre in datos["comodines"]]
272                if datos["comodines"]
273                else []
274            )
275            self.mano = [Carta(c[0], c[1]) for c in datos["mano"]]
276            self.mazo.cartas = [Carta(c[0], c[1]) for c in datos["mazo"]]
277            self._cartas_jugadas = (
278                [Carta(c[0], c[1]) for c in datos["cartas_jugadas"]]
279                if datos["cartas_jugadas"]
280                else None
281            )
282            return True
283        else:
284            return None

155        Funcion jugar cartas:
156            Comprueba si hay cartas seleccionadas y las juega o las descarta.
157        """
158        if (
159            len(self._cartas_seleccionadas) <= self.limite_seleccion

186            for (
187                carta
