 56                juego.mostrar_pagina_juego(screen)

 59                self.seleccionar_cartas([3, 4, 5, 6], juego)
 60            case 7:
 61                juego.mostrar_pagina_juego(screen)
 62                self.boton_jugar(juego)
 63            case 8:
 64                juego.mostrar_pagina_juego(screen)

 66            case 9:
 67                juego.mostrar_pagina_juego(screen)

 71                if Juego.num_transicion:

116class Cursor:
117    def __init__(self, x, y):
118        self.imagen = pygame.image.load("Graficos/cursor.png").convert_alpha()   # Carga la imagen
119        self.x = x
120        self.y = y
121        self.x_final = self.x
122        self.y_final = self.y
123        self.velocidad = 0.2
124    
125    def asignar_posicion_carta(self, carta: int):
126        self.x_final = 590 + 90*(carta-1)
127        self.y_final = 560
128
129    def asignar_posicion(self, posx, posy):
130        self.x_final = posx
131        self.y_final = posy
132    
133    def checkear_pos(self):
134        if self.x_final-1 <= self.x <= self.x_final+1 and self.y_final-1 <= self.y <= self.y_final+1:
135            return True
136        else:
137            return False
138
139    def dibujar(self, screen):
140        self._mover_hacia_destino()
141        screen.blit(self.imagen, (self.x, self.y))
142
143    def _mover_hacia_destino(self):
144        self.x += (self.x_final - self.x) * self.velocidad
145        self.y += (self.y_final - self.y) * self.velocidad
